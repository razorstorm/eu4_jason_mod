newcastle_coal = {
	color = { 0.0 1.0 1.0 }
	is_latent = no
	is_valuable = yes

	rnw_latent_chance = 0 # Kind of on average how many provinces we want to spawn with it

    #How likely it will appear in empty provinces on colonization.
    chance = {
        factor = 0
    }
	
	trigger = {
		OR = {
			development_discounting_tribal = 20
			owner = { innovativeness = 20 }
		}
		if = {
			limit = {
				NOT = { has_province_flag = GER_specific_coal }
				owner = {
					NOT = { has_country_flag = earlier_coal_available }
				}
			}
			provincial_institution_progress = {
				which = enlightenment
				value = 100
			}
			owner = { has_institution = enlightenment }
		}
		else_if = {
			limit = {
				has_province_flag = GER_specific_coal
			}
			provincial_institution_progress = {
				which = manufactories
				value = 100
			}
			owner = {
				has_institution = manufactories
				adm_tech = 21
			}
		}
		else = {
			provincial_institution_progress = {
				which = manufactories
				value = 100
			}
			owner = {
				has_institution = manufactories
				adm_tech = 23
			}
		}
	}
	
	modifier = {
		global_trade_goods_size_modifier = 0.15
	}

	modifier = {
		movement_speed = 0.5
	}
	
	province = {
		local_state_maintenance_modifier = -0.25
        trade_goods_size_modifier = 0.2
	}

	chance = {
		factor = 1
	}
}
