# Tier 2. Core Regions (Admin region, real union subjects, imperial cities, banks, etc)
establish_administrative_region = {
    category = subjects

    require_acceptance = no

    is_visible = {
        ai = no
        FROM = {
            is_subject_of = ROOT
            NOT = { is_subject_of_type = administrative_region }
        }
    }
    is_allowed = {
        ai = no
        FROM = { 
            is_subject_of = ROOT
            NOT = { is_subject_of_type = administrative_region }
        }
    }
    on_accept = {
        create_subject = {
            subject_type = administrative_region
            subject = FROM
        }
    }

    # Create it regardless
    on_decline = {
        create_subject = {
            subject_type = administrative_region
            subject = FROM
        }
    }

    ai_will_do = { # this is a trigger
        always = no
    }
}

    category = subjects

    require_acceptance = no

        ai = no

        }
    }
    is_allowed = {
        ai = no
        FROM = { 
            is_subject_of = ROOT
            OR = {
                NOT = { has_country_flag = free_city }
                NOT = { is_subject_of_type = imperial_city }
            }
        }
    }
    on_accept = {
        create_subject = {
            subject_type = imperial_city
